def fetch_redash(qid):
    r = requests.get(f'{REDASH_BASE_URL}/api/queries/{qid}/results',
                     headers={'Authorization': f'Key {REDASH_API_KEY}'}, timeout=30)
    r.raise_for_status()
    return r.json()['query_result']['data']['rows']

def parse_date(s):
    if not s or str(s).strip() in ('-', '○', '×', '\\-', ''): return None
    s = str(s).strip().split(' ')[0]
    for fmt in ('%Y/%m/%d', '%Y-%m-%d'):
        try: return datetime.strptime(s, fmt)
        except: pass
    return None

def three_month_end():
    d = TODAY + timedelta(days=92)
    nm = d.replace(day=28) + timedelta(days=4)
    return nm - timedelta(days=nm.day)

cutoff = three_month_end()
data_3029 = fetch_redash(3029)
data_3023 = fetch_redash(3023)

metrics_raw = defaultdict(lambda: {'co': [], 'cl': []})
for r in data_3023:
    area = r.get('エリア', '')
    pref = r.get('都道府県', '')
    if not area or area == 'その他': continue
    k = (pref, area)
    co = r.get('チェックアウト数', 0) or 0
    prop = r.get('物件数', 0) or 0
    if prop > 0: metrics_raw[k]['co'].append(co / prop)
    cl = r.get('クリーナー1人あたり完了数', 0) or 0
    if cl > 0: metrics_raw[k]['cl'].append(cl)

def avg_metrics(pref, area):
    m = metrics_raw.get((pref, area), {})
    co_list = m.get('co', [])
    cl_list = m.get('cl', [])
    cop = sum(co_list) / len(co_list) if co_list else None
    clp = max(sum(cl_list) / len(cl_list) if cl_list else 5.3, 5.3)
    return cop, clp

                 and r.get('エリア', '') not in ('', 'その他')])
optimistic = int(round(sum(all_cl[int(len(all_cl)*0.75):]) / max(len(all_cl[int(len(all_cl)*0.75):]), 1))) if all_cl else 25
